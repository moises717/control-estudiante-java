CREATE TABLE secciones (
	ID INTEGER  PRIMARY KEY AUTOINCREMENT,
	NUM_SECCION INTEGER NOT NULL,
        ASIGNATURAS varchar(50)


);

CREATE TABLE estudiantes (
	ID INTEGER PRIMARY KEY AUTOINCREMENT,
	ID_DOCENTE INTEGER NOT NULL,
	ID_SECCION INTEGER NOT NULL,
	NOMBRES varchar(50),
	APELLIDOS varchar(50),
	IMAGEN varchar(100),
        FOREIGN KEY(ID_DOCENTE) REFERENCES docentes(ID),
        FOREIGN KEY(ID_SECCION) REFERENCES secciones(ID)

);

CREATE TABLE notas (
	ID INTEGER PRIMARY KEY AUTOINCREMENT,
	ID_ESTUDIANTE INTEGER,
	NOTA1 INTEGER DEFAULT 0,
	NOTA2 INTEGER DEFAULT 0,
	NOTA3 INTEGER DEFAULT 0,
	EXAMEN INTEGER DEFAULT 0,
	FINALIZADO BOOLEAN DEFAULT false,
	FOREIGN KEY(ID_ESTUDIANTE) REFERENCES estudiantes(ID)
);

CREATE TABLE docentes (
	ID INTEGER PRIMARY KEY AUTOINCREMENT,
	NOMBRES varchar(50),
	APELLIDOS varchar(50)

);
